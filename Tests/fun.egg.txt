fun fibonacci(limit) {

    var a = 0
    var b = 1
    var result = []

    while (a < limit) {

        result = result + [a,]

        var next = a + b

        a = b
        b = next
    }

    return result
}


var fib = fibonacci(50)

fineshow("T9:", fib)
