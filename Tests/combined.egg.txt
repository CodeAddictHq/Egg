fun calculate_order(order) {

    var subtotal = 0

    for (item in order.items) {

        var item_total = item.price * item.quantity

        subtotal = subtotal + item_total

    }


    var discount = 0

    if (subtotal > 1000) {

        discount = subtotal / 10

    } elif (subtotal > 500) {

        discount = 50

    }


    var tax = subtotal / 20

    var final_price = subtotal - discount + tax


    return {
        customer: order.customer,

        items: order.items,

        calculation: {
            subtotal: subtotal,
            discount: discount,
            tax: tax,
            total: final_price,
        },

        valid: final_price > 0 and order.active,

    }

}

fun process_bank(customer) {

    var balance = customer.account.balance

    var transactions = customer.account.transactions

    var history = []


    for (transaction in transactions) {


        if (transaction.type is "deposit") {

            balance = balance + transaction.amount


        } elif (transaction.type is "withdraw") {


            if (balance >_ transaction.amount) {

                balance = balance - transaction.amount

            }

        }


        history = history + [

            {

                type: transaction.type,

                amount: transaction.amount,

                balance: balance,

            },

        ]

    }


    var risk = FALSE


    if (balance < 500) {

        risk = TRUE

    }


    return {

        customer: customer.name,


        account: {

            final_balance: balance,

            transaction_count: history,

            risky: risk,

        },


        active: customer.account.active and balance > 0,

    }

}

fun process_hospital(hospital) {

    var patient_reports = []
    var emergency_count = 0
    var total_cost = 0


    for (department in hospital.departments) {


        for (patient in department.patients) {


            var risk = "LOW"


            var cost = 0



            for (treatment in patient.treatments) {

                cost = cost + treatment.price

            }



            if (patient.vitals.heart_rate > 120 or patient.vitals.temperature > 39) {

                risk = "HIGH"

                emergency_count = emergency_count + 1

            } elif (patient.vitals.heart_rate > 90) {

                risk = "MEDIUM"

            }



            if (patient.insurance.active is TRUE) {

                cost = cost - 100

            }



            total_cost = total_cost + cost



            patient_reports = patient_reports + [

                {

                    name: patient.name,

                    department: department.name,

                    diagnosis: patient.diagnosis,

                    risk: risk,

                    bill: cost,

                    healthy: risk is "LOW",

                },

            ]

        }

    }



    return {


        hospital: hospital.name,


        summary: {

            patients: patient_reports,

            emergency_cases: emergency_count,

            total_income: total_cost,

        },


        operational: hospital.active and emergency_count < 5,

}

fun finance_system(data) {

    var reports = []
    var total = 0

    for (account in data.accounts) {

        var balance = account.balance

        for (transaction in account.transactions) {

            if (transaction.type is "add") {
                balance = balance + transaction.amount
            } else {
                balance = balance - transaction.amount
            }

        }

        total = total + balance

        reports = reports + [{
            user: account.user,
            balance: balance,
            valid: balance > 0,
        }]

    }

    return {
        reports: reports,
        total: total,
    }
}

            balance:1000,
            transactions:[

            ],
        },
    ],
})

fun security_scan(system) {

    var threats = []

    for (server in system.servers) {

        if (server.online and server.security.updated is FALSE) {

            threats = threats + [
                server.name,
            ]

        }

    }

    return {
        threats:threats,
        safe:threats[0] is NONE,
    }

}

fun restaurant(order) {

    var total = 0

    for (item in order.items) {

        total = total + item.price * item.count

    }

    return {
        customer:order.name,
        bill:total,
        paid:total > 0,
    }

}
